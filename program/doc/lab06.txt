Основна частина:
- опис роботи: розтавлення чисел за зростанням.
#include <stdio.h>
# define  A 6//определяет инднкатор как макрос ,дает команду заменять все последоватильные случаи идентификатора.
int main() {
    int b[A] ={7,9,3,5,1,120};
    int i,j,c;
    for ( i = 0; i< A;i++)
     {
        for ( j  =0;j<A-i-2;j++)
         {
            if (b[j] > b[j+1])
            {
                c =b[j];
                b[j] = b[j+1];
                b[j+1] = c;
            }
       }
        printf ("%d\n",c);
    } 
    return 0;
}
