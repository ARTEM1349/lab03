#include <stdio.h>
#include<iostream>
// Определение любых тконстант

//Функция
int main(){

    //переменные
   float USD = (40), EUR = (39), RUB = (0.3f),UAH, x,y,z ;

    // f перед значениям рубля из-за того что 0.3  не целое число , а десятичное поетому стоит команда float  перед переменными

   //переменная которую задает пользыватель
   scanf("%f",&UAH); 

   // расчет значений
    x = UAH/USD;
    y = UAH/EUR;
    z = UAH/RUB;

    // вывод данных 

    //%f / %f = %.4f\n" что именно надо выводить 

    //.4f\n точка обозначает что расчет идет до десятитысячних,  команда \n обозначает что вывод данных идет с новой  строки.

    //UAH, USD, x  значения что мы вписываем в вывод данных.
    
    printf("USD: %f / %f = %.4f\n", UAH, USD, x); 
    printf(" EUR: %f / %f = %.4f\n", UAH, EUR, y);
    printf(" RUB: %f / %f = %.4f\n", UAH, RUB, z);

    return 0;
}
